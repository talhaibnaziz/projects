    } else if(array[x][y]=='#') {

        array[x][y]='#';
        printf("\ndyes");
        ++cnt;
        int a,b,c,d;
        printf("\n\ncount==%d",cnt);

        a=search(x-1,y,cnt);

            return cnt;
        }
    }
}
